Array.diff

function arrayDiff(a, b) {
  if (a.length === 0) return [];
  if (b.length === 0) return a;

  return a.filter((item) => {
    if (b.indexOf(item) === -1) {
      return item;
    }
  });
}

function stray(numbers) {
  return numbers.find((number) => {
    return numbers.indexOf(number) == numbers.lastIndexOf(number)
  });
}
